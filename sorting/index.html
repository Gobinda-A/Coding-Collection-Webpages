Merge Sort


#include <bits/stdc++.h>

using namespace std;

void merge(int arr[], int l, int mid, int r)

{

    int i, j, k;

    int n1 = mid - l + 1;

    int n2 = r - mid;

    int left[n1], right[n2];

    for (i = 0; i < n1; i++)

        left[i] = arr[l + i];

    for (j = 0; j < n2; j++)

        right[j] = arr[mid + 1 + j];

    i = 0;

    j = 0;

    k = l;

    while (i < n1 && j < n2)

    {

        if (left[i] <= right[j])

        {

            arr[k] = left[i];

            i++;

        }

        else

        {

            arr[k] = right[j];

            j++;

        }

        k++;

    }

    while (i < n1)

    {

        arr[k] = left[i];

        i++;

        k++;

    }

    while (j < n2)

    {

        arr[k] = right[j];

        j++;

        k++;

    }

}



void mergesort(int arr[], int l, int r)

{

    if (l < r)

    {

        int mid = l + (r - l) / 2;

        mergesort(arr, l, mid);

        mergesort(arr, mid + 1, r);

        merge(arr, l, mid, r);

    }

}



int main()

{

    int arr[] = {4, 7, 2, 9, 1, 3, 8, 5, 6};

    int size = sizeof(arr) / sizeof(arr[0]);

    mergesort(arr, 0, size - 1);

    for (int i = 0; i < size; i++)

    {

        cout << arr[i] << " ";

    }

    return 0;

}
